cmake_minimum_required(VERSION 3.5)

project(Noma LANGUAGES CXX)

set(CMAKE_CXX_STANDARD 17)
set(CMAKE_CXX_STANDARD_REQUIRED ON)

set(Main main.cpp)
set(AllFiles
      noma.cpp
      PhoneBook.cpp
      Contact.cpp
      noma.hpp
      PhoneBook.hpp
      Contact.hpp)
set(Testers 
    PhoneBook.cpp
    PhoneBook_unittest.cpp)

add_executable(noma ${Main} ${AllFiles})
if (MSVC)
  # warning level 4
  target_compile_options(noma
        /W4)
else()
  # additional warnings
  target_compile_options(noma PUBLIC
        -Wextra
        -Werror
        -Wall
        -Wpedantic)
endif()

#add_executable(nomaTDD ${AllFiles} ${Testers}...

include(GNUInstallDirs)
install(TARGETS noma
    LIBRARY DESTINATION ${CMAKE_INSTALL_LIBDIR}
    RUNTIME DESTINATION ${CMAKE_INSTALL_BINDIR}
)
